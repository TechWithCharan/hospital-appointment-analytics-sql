================================================================================
                                RELATIONSHIPS
================================================================================

1. PATIENTS (1) -----> (N) APPOINTMENTS
   - One patient can have multiple appointments
   - Each appointment belongs to exactly one patient
   - Foreign Key: appointments.patient_id -> patients.patient_id

2. DOCTORS (1) -----> (N) APPOINTMENTS
   - One doctor can have multiple appointments
   - Each appointment is handled by exactly one doctor
   - Foreign Key: appointments.doctor_id -> doctors.doctor_id

================================================================================
                                DATA TYPES & CONSTRAINTS
================================================================================

PATIENTS TABLE:
- patient_id: Primary Key, Auto-increment
- age: CHECK constraint (0 < age < 150)
- gender: CHECK constraint ('Male', 'Female')
- insurance_type: CHECK constraint ('Private', 'Public')
- has_chronic_disease: CHECK constraint ('Yes', 'No')
- diabetic: CHECK constraint ('Yes', 'No')
- income_level: CHECK constraint ('Low', 'Medium', 'High')

DOCTORS TABLE:
- doctor_id: Primary Key, Auto-increment
- experience_years: CHECK constraint (>= 0)
- consultation_fee: CHECK constraint (>= 0)
- rating: CHECK constraint (0 <= rating <= 5)
- is_active: CHECK constraint ('Yes', 'No')
- emergency_available: CHECK constraint ('Yes', 'No')

APPOINTMENTS TABLE:
- appointment_id: Primary Key, Auto-increment
- status: CHECK constraint ('Attended', 'No-Show')
- appointment_type: CHECK constraint ('Routine', 'Emergency', 'Follow-up')
- consultation_duration: CHECK constraint (> 0)
- waiting_time_minutes: CHECK constraint (>= 0)
- feedback_rating: CHECK constraint (1 <= rating <= 5)
- payment_status: CHECK constraint ('Paid', 'Pending')
- prescription_given: CHECK constraint ('Yes', 'No')
- lab_tests_ordered: CHECK constraint ('Yes', 'No')
- follow_up_required: CHECK constraint ('Yes', 'No')
